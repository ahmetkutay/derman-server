Auth API Documentation

Login

POST /v1/auth/

Request Body:

{
    "username": "user123",
    "password": "password123"
}

Response:

{
    "accessToken": "yourAccessToken",
    "refreshedToken": "yourRefreshToken"
}

Register

POST /v1/auth/register

Request Body:

{
    "username": "user123",
    "first_name": "John",
    "last_name": "Doe",
    "email": "john.doe@example.com",
    "password": "password123",
    "phone": "1234567890",
    "birthdate": "1990-01-01",
    "gender": "male"
}

Response:

{
    "user": "userId"
}

Logout

POST /v1/auth/logout

Request Header:

Authorization: Bearer <token>

Response:

{
    "message": "Logged out successfully"
}

Verification

POST /v1/auth/verification

Request Body:

{
    "token": "verificationToken",
    "id": "userId"
}

Response:

"Verified"
